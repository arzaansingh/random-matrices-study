%% style/colors.tex
%% Locked color palette for the thesis. Six colors total, defined once,
%% inherited everywhere. Any new color is a signal the design has drifted;
%% pause and reconsider before adding one.

% Sage green: first-use vocabulary highlight, theorem/definition counter chip.
% Inspired by Joseph Genzer's thesis convention.
\definecolor{thesisVocabBg}{HTML}{D4E8C8}

% Slate navy: definition environment left rule.
\definecolor{thesisDef}{HTML}{3A5A78}

% Charcoal: theorem/proposition/lemma/corollary left rule. Restrained.
\definecolor{thesisThm}{HTML}{2D2D2D}

% Warm cream: subtle background tint for the intuition environment.
\definecolor{thesisIntuitionBg}{HTML}{FBF3D9}

% Muted blue: hyperlinks, citations, internal cross-references.
\definecolor{thesisLink}{HTML}{3D7BAA}

% Warm gray: figure caption rule, code listing border, secondary accents.
\definecolor{thesisAccent}{HTML}{8B8378}

% Convenience: highlight color for the soul package (used by \vocab and the
% theorem-counter highlight).
\colorlet{thesisHL}{thesisVocabBg}
